## Description
어떤 게임에는 붕대 감기라는 기술이 있습니다.

붕대 감기는 t초 동안 붕대를 감으면서 1초마다 x만큼의 체력을 회복합니다. t초 연속으로 붕대를 감는 데 성공한다면 y만큼의 체력을 추가로 회복합니다. 게임 캐릭터에는 최대 체력이 존재해 현재 체력이 최대 체력보다 커지는 것은 불가능합니다.

기술을 쓰는 도중 몬스터에게 공격을 당하면 기술이 취소되고, 공격을 당하는 순간에는 체력을 회복할 수 없습니다. 몬스터에게 공격당해 기술이 취소당하거나 기술이 끝나면 그 즉시 붕대 감기를 다시 사용하며, 연속 성공 시간이 0으로 초기화됩니다.

몬스터의 공격을 받으면 정해진 피해량만큼 현재 체력이 줄어듭니다. 이때, 현재 체력이 0 이하가 되면 캐릭터가 죽으며 더 이상 체력을 회복할 수 없습니다.

당신은 붕대감기 기술의 정보, 캐릭터가 가진 최대 체력과 몬스터의 공격 패턴이 주어질 때 캐릭터가 끝까지 생존할 수 있는지 궁금합니다.

붕대 감기 기술의 시전 시간, 1초당 회복량, 추가 회복량을 담은 1차원 정수 배열 bandage와 최대 체력을 의미하는 정수 health, 몬스터의 공격 시간과 피해량을 담은 2차원 정수 배열 attacks가 매개변수로 주어집니다. 모든 공격이 끝난 직후 남은 체력을 return 하도록 solution 함수를 완성해 주세요. 만약 몬스터의 공격을 받고 캐릭터의 체력이 0 이하가 되어 죽는다면 -1을 return 해주세요.

## 제한사항
bandage는 [시전 시간, 초당 회복량, 추가 회복량] 형태의 길이가 3인 정수 배열입니다.
1 ≤ 시전 시간 = t ≤ 50
1 ≤ 초당 회복량 = x ≤ 100
1 ≤ 추가 회복량 = y ≤ 100
1 ≤ health ≤ 1,000
1 ≤ attacks의 길이 ≤ 100
attacks[i]는 [공격 시간, 피해량] 형태의 길이가 2인 정수 배열입니다.
attacks는 공격 시간을 기준으로 오름차순 정렬된 상태입니다.
attacks의 공격 시간은 모두 다릅니다.
1 ≤ 공격 시간 ≤ 1,000
1 ≤ 피해량 ≤ 100

## SOLUTION

```tsx
function solution(bandage, health, attacks) {
    // 캐릭터의 현재 체력을 health 변수로 초기화합니다.
    let currentHealth = health;

    // bandage 배열을 구조 분해 할당을 통해 각 변수에 할당합니다.
    // t는 붕대 감는 시간, x는 초당 회복량, y는 추가 회복량입니다.
    const [t, x, y] = bandage;

    // 연속으로 붕대를 감는 시간을 추적하는 변수입니다.
    let continuousTime = 0;

    // 처리해야 할 다음 공격의 인덱스입니다.
    let attackIndex = 0;

    // 게임 시간을 추적하는 변수입니다.
    let time = 0;

    // 무한 루프를 시작합니다.
    // 이 루프는 break 문을 만날 때까지, 혹은 return 문을 통해 함수가 종료될 때까지 실행됩니다.
    while (true) {
        // 만약 현재 시간이 공격 시간과 같다면 공격을 처리합니다.
        if (attackIndex < attacks.length && time === attacks[attackIndex][0]) {
            // 현재 체력에서 공격으로 인한 피해량을 빼줍니다.
            currentHealth -= attacks[attackIndex][1];
            // 다음 공격으로 인덱스를 이동시킵니다.
            attackIndex++;
            // 공격을 받았으므로 연속 시간을 0으로 초기화합니다.
            continuousTime = 0;
        } else {
            // 공격이 없을 경우, 초당 회복량만큼 체력을 회복합니다.
            if (continuousTime < t) {
                currentHealth += x;
                // 연속 성공 시간을 증가시킵니다.
                continuousTime++;
            }
            // 만약 연속 성공 시간이 붕대 감는 시간 t에 도달했다면 추가 회복을 적용합니다.
            if (continuousTime === t) {
                currentHealth += y;
                // 추가 회복을 적용한 후에는 연속 성공 시간을 0으로 초기화합니다.
                continuousTime = 0;
            }
        }

        // 체력이 최대 체력을 초과하지 않도록 합니다.
        if (currentHealth > health) {
            currentHealth = health;
        }

        // 체력이 0 이하가 되면 캐릭터가 죽었으므로 -1을 반환하고 함수를 종료합니다.
        if (currentHealth <= 0) {
            return -1;
        }

        // 모든 공격이 끝났고 연속 성공 시간이 0이면 루프를 종료합니다.
        if (attackIndex >= attacks.length && continuousTime === 0) {
            break;
        }

        // 시간을 1초 증가시킵니다.
        time++;
    }

    // 모든 공격을 처리한 후 남은 체력을 반환합니다.
    return currentHealth;
}
```
